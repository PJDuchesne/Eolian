ORG	$1000

MAIN

MOV	#$100,R10	; R10 will have 0x0100

SUB	R10,R12		; R12 will have 0x0400

MOV	R10,R14		; Should move 0x0100 to R14
SWPB	R14		; R14 will have 0x0001

END	MAIN
